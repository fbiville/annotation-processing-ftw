  <version>3.2</version>
  <type>maven-plugin</type>
</dependency>

Apache Buildr

'org.apache.maven.plugins:maven-compiler-plugin:maven-plugin:3.2'

Apache Ant

<dependency org="org.apache.maven.plugins" name="maven-compiler-plugin" rev="3.2">

</dependency>

Groovy Grape
